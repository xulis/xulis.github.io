```golang
package main
import "fmt"
func main() {
	//A 只对1,B只对2,C都对,D都不对
	SUM := 45
	for A := 0; A <= SUM; A++ {
		for B := 0; B <= SUM; B++ {
			for C := 0; C <= SUM; C++ {
				for D := 0; D <= SUM; D++ {
					if A+C == 33 && B+C == 28 && A+B+C+D == SUM {
						fmt.Printf("只对第一题:%d\n只对第二题:%d\n两题都做对:%d\n两题都不对:%d\n", A, B, C, D)
						fmt.Println("------")
					}
				}
			}
		}
	}
}
```
